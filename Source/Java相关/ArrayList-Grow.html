        // oldCapacityСИ║ТЌДт«╣жЄЈ№╝їnewCapacityСИ║Тќ░т«╣жЄЈ
        int oldCapacity = elementData.length;

С╗јСИіжЮб grow() Тќ╣Т│ЋТ║љуаЂТѕЉС╗гуЪЦжЂЊ№╝џ тдѓТъюТќ░т«╣жЄЈтцДС║ј MAX_ARRAY_SIZE,У┐ЏтЁЦ(ТЅДУАї) hugeCapacity() Тќ╣Т│ЋТЮЦТ»ћУЙЃ minCapacity тњї MAX_ARRAY_SIZE№╝їтдѓТъюminCapacityтцДС║јТюђтцДт«╣жЄЈ№╝їтѕЎТќ░т«╣жЄЈтѕЎСИ║Integer.MAX_VALUE№╝їтљдтѕЎ№╝їТќ░т«╣жЄЈтцДт░ЈтѕЎСИ║ MAX_ARRAY_SIZE тЇ│СИ║ Integer.MAX_VALUE - 8сђѓ

    private static int hugeCapacity(int minCapacity) {
        if (minCapacity < 0) // overflow
            throw new OutOfMemoryError();
        //т»╣minCapacityтњїMAX_ARRAY_SIZEУ┐ЏУАїТ»ћУЙЃ
        //УІЦminCapacityтцД№╝їт░єInteger.MAX_VALUEСйюСИ║Тќ░ТЋ░у╗ёуџётцДт░Ј
        //УІЦMAX_ARRAY_SIZEтцД№╝їт░єMAX_ARRAY_SIZEСйюСИ║Тќ░ТЋ░у╗ёуџётцДт░Ј
        //MAX_ARRAY_SIZE = Integer.MAX_VALUE - 8;
        return (minCapacity > MAX_ARRAY_SIZE) ?
            Integer.MAX_VALUE :
            MAX_ARRAY_SIZE;
    }
