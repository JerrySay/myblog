# final тЁ│жћ«тГЌ

finalтЁ│жћ«тГЌСИ╗УдЂућетюеСИЅСИфтю░Тќ╣№╝џтЈўжЄЈсђЂТќ╣Т│ЋсђЂу▒╗сђѓ

  1. т»╣С║јСИђСИфfinalтЈўжЄЈ№╝їтдѓТъюТў»тЪ║ТюгТЋ░ТЇ«у▒╗тъІуџётЈўжЄЈ№╝їтѕЎтЁХТЋ░тђ╝СИђТЌдтюетѕЮтДІтїќС╣ІтљјСЙ┐СИЇУЃйТЏ┤Тћ╣№╝ЏтдѓТъюТў»т╝Ћућеу▒╗тъІуџётЈўжЄЈ№╝їтѕЎтюет»╣тЁХтѕЮтДІтїќС╣ІтљјСЙ┐СИЇУЃйтєЇУ«ЕтЁХТїЄтљЉтЈдСИђСИфт»╣У▒Асђѓ

  2. тйЊућеfinalС┐«жЦ░СИђСИфу▒╗ТЌХ№╝їУАеТўјУ┐ЎСИфу▒╗СИЇУЃйУбФу╗ДТЅ┐сђѓfinalу▒╗СИГуџёТЅђТюЅТѕљтЉўТќ╣Т│ЋжЃйС╝џУбФжџљт╝Јтю░ТїЄт«џСИ║finalТќ╣Т│Ћсђѓ

  3. Сй┐ућеfinalТќ╣Т│ЋуџётјЪтЏаТюЅСИцСИфсђѓуггСИђСИфтјЪтЏаТў»ТііТќ╣Т│ЋжћЂт«џ№╝їС╗Цжў▓С╗╗СйЋу╗ДТЅ┐у▒╗С┐«Тћ╣т«ЃуџётљФС╣Ѕ№╝ЏуггС║їСИфтјЪтЏаТў»ТЋѕујЄсђѓтюеТЌЕТюЪуџёJavaт«ъуј░уЅѕТюгСИГ№╝їС╝џт░єfinalТќ╣Т│ЋУйгСИ║тєЁтхїУ░ЃућесђѓСйєТў»тдѓТъюТќ╣Т│ЋУ┐ЄС║јт║ътцД№╝їтЈ»УЃйуюІСИЇтѕ░тєЁтхїУ░ЃућетИдТЮЦуџёС╗╗СйЋТђДУЃйТЈљтЇЄ№╝ѕуј░тюеуџёJavaуЅѕТюгти▓у╗ЈСИЇжюђУдЂСй┐ућеfinalТќ╣Т│ЋУ┐ЏУАїУ┐ЎС║ЏС╝ўтїќС║є№╝Ѕсђѓу▒╗СИГТЅђТюЅуџёprivateТќ╣Т│ЋжЃйжџљт╝Јтю░ТїЄт«џСИ║finalсђѓ

# static тЁ│жћ«тГЌ

static тЁ│жћ«тГЌСИ╗УдЂТюЅС╗ЦСИІтЏЏуДЇСй┐ућетю║ТЎ»№╝џ

  1. С┐«жЦ░ТѕљтЉўтЈўжЄЈтњїТѕљтЉўТќ╣Т│Ћ: УбФ static С┐«жЦ░уџёТѕљтЉўт▒ъС║ју▒╗№╝їСИЇт▒ъС║јтЇЋСИфУ┐ЎСИфу▒╗уџёТЪљСИфт»╣У▒А№╝їУбФу▒╗СИГТЅђТюЅт»╣У▒АтЁ▒С║Ф№╝їтЈ»С╗Цт╣ХСИћт╗║У««жђџУ┐Єу▒╗тљЇУ░ЃућесђѓУбФstatic тБ░ТўјуџёТѕљтЉўтЈўжЄЈт▒ъС║јжЮЎТђЂТѕљтЉўтЈўжЄЈ№╝їжЮЎТђЂтЈўжЄЈ тГўТћЙтюе Java тєЁтГўтї║тЪЪуџёТќ╣Т│Ћтї║сђѓУ░ЃућеТа╝т╝Ј№╝џу▒╗тљЇ.жЮЎТђЂтЈўжЄЈтљЇ у▒╗тљЇ.жЮЎТђЂТќ╣Т│ЋтљЇ()
  2. жЮЎТђЂС╗БуаЂтЮЌ: жЮЎТђЂС╗БуаЂтЮЌт«џС╣Ѕтюеу▒╗СИГТќ╣Т│Ћтцќ, жЮЎТђЂС╗БуаЂтЮЌтюежЮъжЮЎТђЂС╗БуаЂтЮЌС╣ІтЅЇТЅДУАї(жЮЎТђЂС╗БуаЂтЮЌРђћ>жЮъжЮЎТђЂС╗БуаЂтЮЌРђћ>ТъёжђаТќ╣Т│Ћ)сђѓ У»Цу▒╗СИЇу«АтѕЏт╗║тцџт░Љт»╣У▒А№╝їжЮЎТђЂС╗БуаЂтЮЌтЈфТЅДУАїСИђТгА.
  3. жЮЎТђЂтєЁжЃеу▒╗№╝ѕstaticС┐«жЦ░у▒╗уџёУ»ЮтЈфУЃйС┐«жЦ░тєЁжЃеу▒╗№╝Ѕ№╝џ жЮЎТђЂтєЁжЃеу▒╗СИјжЮъжЮЎТђЂтєЁжЃеу▒╗С╣ІжЌ┤тГўтюеСИђСИфТюђтцДуџётї║тѕФ: жЮъжЮЎТђЂтєЁжЃеу▒╗тюеу╝ќУ»Љт«їТѕљС╣ІтљјС╝џжџљтљФтю░С┐ЮтГўуЮђСИђСИфт╝Ћуће№╝їУ»Цт╝ЋућеТў»ТїЄтљЉтѕЏт╗║т«ЃуџётцќтЏ┤у▒╗№╝їСйєТў»жЮЎТђЂтєЁжЃеу▒╗тЇ┤Т▓АТюЅсђѓТ▓АТюЅУ┐ЎСИфт╝Ћућет░▒ТёЈтЉ│уЮђ№╝џ1. т«ЃуџётѕЏт╗║Тў»СИЇжюђУдЂСЙЮУхќтцќтЏ┤у▒╗уџётѕЏт╗║сђѓ2. т«ЃСИЇУЃйСй┐ућеС╗╗СйЋтцќтЏ┤у▒╗уџёжЮъstaticТѕљтЉўтЈўжЄЈтњїТќ╣Т│Ћсђѓ
  4. жЮЎТђЂт»╝тїЁ(ућеТЮЦт»╝тЁЦу▒╗СИГуџёжЮЎТђЂУхёТ║љ№╝ї1.5С╣ІтљјуџёТќ░уЅ╣ТђД): Та╝т╝ЈСИ║№╝џimport static У┐ЎСИцСИфтЁ│жћ«тГЌУ┐ъућетЈ»С╗ЦТїЄт«џт»╝тЁЦТЪљСИфу▒╗СИГуџёТїЄт«џжЮЎТђЂУхёТ║љ№╝їт╣ХСИћСИЇжюђУдЂСй┐ућеу▒╗тљЇУ░Ѓућеу▒╗СИГжЮЎТђЂТѕљтЉў№╝їтЈ»С╗ЦуЏ┤ТјЦСй┐ућеу▒╗СИГжЮЎТђЂТѕљтЉўтЈўжЄЈтњїТѕљтЉўТќ╣Т│Ћсђѓ

# this тЁ│жћ«тГЌ

thisтЁ│жћ«тГЌућеС║јт╝Ћућеу▒╗уџётйЊтЅЇт«ъСЙІсђѓ СЙІтдѓ№╝џ

class Manager {
    Employees[] employees;
     
    void manageEmployees() {
        int totalEmp = this.employees.length;
        System.out.println("Total employees: " + totalEmp);
        this.report();
    }
     
    void report() { }
}

тюеСИіжЮбуџёуц║СЙІСИГ№╝їthisтЁ│жћ«тГЌућеС║јСИцСИфтю░Тќ╣№╝џ

  • this.employees.length№╝џУ«┐жЌ«у▒╗ManagerуџётйЊтЅЇт«ъСЙІуџётЈўжЄЈсђѓ
  • this.report№╝ѕ№╝Ѕ№╝џУ░Ѓућеу▒╗ManagerуџётйЊтЅЇт«ъСЙІуџёТќ╣Т│Ћсђѓ

ТГцтЁ│жћ«тГЌТў»тЈ»жђЅуџё№╝їУ┐ЎТёЈтЉ│уЮђтдѓТъюСИіжЮбуџёуц║СЙІтюеСИЇСй┐ућеТГцтЁ│жћ«тГЌуџёТЃЁтєхСИІУАеуј░уЏИтљїсђѓ СйєТў»№╝їСй┐ућеТГцтЁ│жћ«тГЌтЈ»УЃйС╝џСй┐С╗БуаЂТЏ┤ТўЊУ»╗ТѕќТўЊТЄѓсђѓ

# super тЁ│жћ«тГЌ

superтЁ│жћ«тГЌућеС║јС╗јтГљу▒╗У«┐жЌ«уѕХу▒╗уџётЈўжЄЈтњїТќ╣Т│Ћсђѓ СЙІтдѓ№╝џ

public class Super {
    protected int number;
     
    protected showNumber() {
        System.out.println("number = " + number);
    }
}
 
public class Sub extends Super {
    void bar() {
        super.number = 10;
        super.showNumber();
    }
}

тюеСИіжЮбуџёСЙІтГљСИГ№╝їSub у▒╗У«┐жЌ«уѕХу▒╗ТѕљтЉўтЈўжЄЈ number т╣ХУ░ЃућетЁХтЁХуѕХу▒╗ Super уџё showNumber№╝ѕ№╝Ѕ Тќ╣Т│Ћсђѓ

Сй┐уће this тњї super УдЂТ│еТёЈуџёжЌ«жбў№╝џ

  • super У░ЃућеуѕХу▒╗СИГуџётЁХС╗ќТъёжђаТќ╣Т│ЋТЌХ№╝їУ░ЃућеТЌХУдЂТћЙтюеТъёжђаТќ╣Т│ЋуџёждќУАї№╝Ђthis У░ЃућеТюгу▒╗СИГуџётЁХС╗ќТъёжђаТќ╣Т│ЋТЌХ№╝їС╣ЪУдЂТћЙтюеждќУАїсђѓ
  • thisсђЂsuperСИЇУЃйућетюеstaticТќ╣Т│ЋСИГсђѓ

у«ђтЇЋУДБжЄіСИђСИІ№╝џ

УбФ static С┐«жЦ░уџёТѕљтЉўт▒ъС║ју▒╗№╝їСИЇт▒ъС║јтЇЋСИфУ┐ЎСИфу▒╗уџёТЪљСИфт»╣У▒А№╝їУбФу▒╗СИГТЅђТюЅт»╣У▒АтЁ▒С║ФсђѓУђї this С╗БУАет»╣Тюгу▒╗т»╣У▒Ауџёт╝Ћуће№╝їТїЄтљЉТюгу▒╗т»╣У▒А№╝ЏУђї super С╗БУАет»╣уѕХу▒╗т»╣У▒Ауџёт╝Ћуће№╝їТїЄтљЉуѕХу▒╗т»╣У▒А№╝ЏТЅђС╗Ц№╝ї thisтњїsuperТў»т▒ъС║јт»╣У▒АУїЃуЋ┤уџёСИюУЦ┐№╝їУђїжЮЎТђЂТќ╣Т│ЋТў»т▒ъС║ју▒╗УїЃуЋ┤уџёСИюУЦ┐сђѓ
