import java.util.Stack;
 
public class Solution {
    Stack<Integer> stack1 = new Stack<Integer>();
    Stack<Integer> stack2 = new Stack<Integer>();
     
    //тйЊТЅДУАїpushТЊЇСйюТЌХ№╝їт░єтЁЃу┤аТи╗тіатѕ░stack1
    public void push(int node) {
        stack1.push(node);
    }
     
    public int pop() {
        //тдѓТъюСИцСИфжўЪтѕЌжЃйСИ║уЕ║тѕЎТіЏтЄ║т╝ѓтИИ,У»┤ТўјућеТѕиТ▓АТюЅpushУ┐ЏС╗╗СйЋтЁЃу┤а
        if(stack1.empty()&&stack2.empty()){
            throw new RuntimeException("Queue is empty!");
        }
        //тдѓТъюstack2СИЇСИ║уЕ║уЏ┤ТјЦт»╣stack2ТЅДУАїpopТЊЇСйю№╝ї
        if(stack2.empty()){
            while(!stack1.empty()){
                //т░єstack1уџётЁЃу┤аТїЅтљјУ┐ЏтЁѕтЄ║pushУ┐Џstack2жЄїжЮб
                stack2.push(stack1.pop());
            }
        }
          return stack2.pop();
    }
}

import java.util.ArrayList;
import java.util.Stack;
//У┐ЎжЂЊжбўТ▓АТЃ│тЄ║ТЮЦ№╝їтЈѓУђЃС║єAliasтљїтГдуџёуГћТАѕ№╝џhttps://www.nowcoder.com/questionTerminal/d77d11405cc7470d82554cb392585106
public class Solution {
    public boolean IsPopOrder(int [] pushA,int [] popA) {
        if(pushA.length == 0 || popA.length == 0)
            return false;
        Stack<Integer> s = new Stack<Integer>();
        //ућеС║јТаЄУ»єт╝╣тЄ║т║ЈтѕЌуџёСйЇуй«
        int popIndex = 0;
        for(int i = 0; i< pushA.length;i++){
            s.push(pushA[i]);
            //тдѓТъюТаѕСИЇСИ║уЕ║№╝їСИћТаѕжАХтЁЃу┤ауГЅС║јт╝╣тЄ║т║ЈтѕЌ
            while(!s.empty() &&s.peek() == popA[popIndex]){
                //тЄ║Таѕ
                s.pop();
                //т╝╣тЄ║т║ЈтѕЌтљЉтљјСИђСйЇ
                popIndex++;
            }
        }
        return s.empty();
    }
}
