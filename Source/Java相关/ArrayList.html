package java.util;

import java.util.function.Consumer;
import java.util.function.Predicate;
import java.util.function.UnaryOperator;


public class ArrayList<E> extends AbstractList<E>
        implements List<E>, RandomAccess, Cloneable, java.io.Serializable
{
    private static final long serialVersionUID = 8683452581122892189L;

    /**
     * ж╗ўУ«цтѕЮтДІт«╣жЄЈтцДт░Ј
     */
    private static final int DEFAULT_CAPACITY = 10;

    /**
     * уЕ║ТЋ░у╗ё№╝ѕућеС║јуЕ║т«ъСЙІ№╝Ѕсђѓ
     */
    private static final Object[] EMPTY_ELEMENTDATA = {};

     //ућеС║јж╗ўУ«цтцДт░ЈуЕ║т«ъСЙІуџётЁ▒С║ФуЕ║ТЋ░у╗ёт«ъСЙІсђѓ
      //ТѕЉС╗гТііт«ЃС╗јEMPTY_ELEMENTDATAТЋ░у╗ёСИГтї║тѕєтЄ║ТЮЦ№╝їС╗ЦуЪЦжЂЊтюеТи╗тіауггСИђСИфтЁЃу┤аТЌХт«╣жЄЈжюђУдЂтбътіатцџт░Љсђѓ
    private static final Object[] DEFAULTCAPACITY_EMPTY_ELEMENTDATA = {};

    /**
     * С┐ЮтГўArrayListТЋ░ТЇ«уџёТЋ░у╗ё
     */
    transient Object[] elementData; // non-private to simplify nested class access

    /**
     * ArrayList ТЅђтїЁтљФуџётЁЃу┤аСИфТЋ░
     */
    private int size;

    /**
     * тИдтѕЮтДІт«╣жЄЈтЈѓТЋ░уџёТъёжђатЄйТЋ░сђѓ№╝ѕућеТѕиУЄфти▒ТїЄт«џт«╣жЄЈ№╝Ѕ
     */
    public ArrayList(int initialCapacity) {
        if (initialCapacity > 0) {
            //тѕЏт╗║initialCapacityтцДт░ЈуџёТЋ░у╗ё
            this.elementData = new Object[initialCapacity];
        } else if (initialCapacity == 0) {
            //тѕЏт╗║уЕ║ТЋ░у╗ё
            this.elementData = EMPTY_ELEMENTDATA;
        } else {
            throw new IllegalArgumentException("Illegal Capacity: "+
                                               initialCapacity);
        }
    }

    /**
     *ж╗ўУ«цТъёжђатЄйТЋ░№╝їDEFAULTCAPACITY_EMPTY_ELEMENTDATA СИ║0.тѕЮтДІтїќСИ║10№╝їС╣Ът░▒Тў»У»┤тѕЮтДІтЁХт«ъТў»уЕ║ТЋ░у╗ё тйЊТи╗тіауггСИђСИфтЁЃу┤ауџёТЌХтђЎТЋ░у╗ёт«╣жЄЈТЅЇтЈўТѕљ10
     */
    public ArrayList() {
        this.elementData = DEFAULTCAPACITY_EMPTY_ELEMENTDATA;
    }

    /**
     * ТъёжђаСИђСИфтїЁтљФТїЄт«џжЏєтљѕуџётЁЃу┤ауџётѕЌУАе№╝їТїЅуЁДт«ЃС╗гућ▒жЏєтљѕуџёУ┐ГС╗БтЎеУ┐ћтЏъуџёжА║т║Јсђѓ
     */
    public ArrayList(Collection<? extends E> c) {
        //
        elementData = c.toArray();
        //тдѓТъюТїЄт«џжЏєтљѕтЁЃу┤аСИфТЋ░СИЇСИ║0
        if ((size = elementData.length) != 0) {
            // c.toArray тЈ»УЃйУ┐ћтЏъуџёСИЇТў»Objectу▒╗тъІуџёТЋ░у╗ёТЅђС╗ЦтіаСИіСИІжЮбуџёУ»ГтЈЦућеС║јтѕцТќГ№╝ї
            //У┐ЎжЄїућетѕ░С║єтЈЇт░ёжЄїжЮбуџёgetClass()Тќ╣Т│Ћ
            if (elementData.getClass() != Object[].class)
                elementData = Arrays.copyOf(elementData, size, Object[].class);
        } else {
            // ућеуЕ║ТЋ░у╗ёС╗БТЏ┐
            this.elementData = EMPTY_ELEMENTDATA;
        }
    }

    /**
     * С┐«Тћ╣У┐ЎСИфArrayListт«ъСЙІуџёт«╣жЄЈТў»тѕЌУАеуџётйЊтЅЇтцДт░Јсђѓ т║ћућеуеІт║ЈтЈ»С╗ЦСй┐ућеТГцТЊЇСйюТЮЦТюђт░ЈтїќArrayListт«ъСЙІуџётГўтѓесђѓ 
     */
    public void trimToSize() {
        modCount++;
        if (size < elementData.length) {
            elementData = (size == 0)
              ? EMPTY_ELEMENTDATA
              : Arrays.copyOf(elementData, size);
        }
    }
//СИІжЮбТў»ArrayListуџёТЅЕт«╣Тю║тѕХ
//ArrayListуџёТЅЕт«╣Тю║тѕХТЈљжФўС║єТђДУЃй№╝їтдѓТъюТ»ЈТгАтЈфТЅЕтЁЁСИђСИф№╝ї
//жѓБС╣ѕжбЉу╣ЂуџёТЈњтЁЦС╝џт»╝УЄ┤жбЉу╣ЂуџёТІиУ┤Ю№╝їжЎЇСйјТђДУЃй№╝їУђїArrayListуџёТЅЕт«╣Тю║тѕХжЂ┐тЁЇС║єУ┐ЎуДЇТЃЁтєхсђѓ
    /**
     * тдѓТюЅт┐ЁУдЂ№╝їтбътіаТГцArrayListт«ъСЙІуџёт«╣жЄЈ№╝їС╗ЦуА«С┐Ют«ЃУЄ│т░ЉУЃйт«╣у║│тЁЃу┤ауџёТЋ░жЄЈ
     * @param   minCapacity   ТЅђжюђуџёТюђт░Јт«╣жЄЈ
     */
    public void ensureCapacity(int minCapacity) {
        int minExpand = (elementData != DEFAULTCAPACITY_EMPTY_ELEMENTDATA)
            // any size if not default element table
            ? 0
            // larger than default for default empty table. It's already
            // supposed to be at default size.
            : DEFAULT_CAPACITY;

        if (minCapacity > minExpand) {
            ensureExplicitCapacity(minCapacity);
        }
    }
   //тЙЌтѕ░Тюђт░ЈТЅЕт«╣жЄЈ
    private void ensureCapacityInternal(int minCapacity) {
        if (elementData == DEFAULTCAPACITY_EMPTY_ELEMENTDATA) {
              // УјитЈќж╗ўУ«цуџёт«╣жЄЈтњїС╝атЁЦтЈѓТЋ░уџёУЙЃтцДтђ╝
            minCapacity = Math.max(DEFAULT_CAPACITY, minCapacity);
        }

        ensureExplicitCapacity(minCapacity);
    }
  //тѕцТќГТў»тљджюђУдЂТЅЕт«╣
    private void ensureExplicitCapacity(int minCapacity) {
        modCount++;

        // overflow-conscious code
        if (minCapacity - elementData.length > 0)
            //У░ЃућеgrowТќ╣Т│ЋУ┐ЏУАїТЅЕт«╣№╝їУ░ЃућеТГцТќ╣Т│ЋС╗БУАети▓у╗Јт╝ђтДІТЅЕт«╣С║є
            grow(minCapacity);
    }

    /**
     * УдЂтѕєжЁЇуџёТюђтцДТЋ░у╗ётцДт░Ј
     */
    private static final int MAX_ARRAY_SIZE = Integer.MAX_VALUE - 8;

    /**
     * ArrayListТЅЕт«╣уџёТаИт┐ЃТќ╣Т│Ћсђѓ
     */
    private void grow(int minCapacity) {
        // oldCapacityСИ║ТЌДт«╣жЄЈ№╝їnewCapacityСИ║Тќ░т«╣жЄЈ
        int oldCapacity = elementData.length;
        //т░єoldCapacity тЈ│уД╗СИђСйЇ№╝їтЁХТЋѕТъюуЏИтйЊС║јoldCapacity /2№╝ї
        //ТѕЉС╗гуЪЦжЂЊСйЇУ┐љу«ЌуџёжђЪт║дУ┐юУ┐ют┐ФС║јТЋ┤жЎцУ┐љу«Ќ№╝їТЋ┤тЈЦУ┐љу«Ќт╝Јуџёу╗ЊТъют░▒Тў»т░єТќ░т«╣жЄЈТЏ┤Тќ░СИ║ТЌДт«╣жЄЈуџё1.5тђЇ№╝ї
        int newCapacity = oldCapacity + (oldCapacity >> 1);
        //уёХтљјТБђТЪЦТќ░т«╣жЄЈТў»тљдтцДС║јТюђт░ЈжюђУдЂт«╣жЄЈ№╝їУІЦУ┐ўТў»т░ЈС║јТюђт░ЈжюђУдЂт«╣жЄЈ№╝їжѓБС╣ѕт░▒ТііТюђт░ЈжюђУдЂт«╣жЄЈтйЊСйюТЋ░у╗ёуџёТќ░т«╣жЄЈ№╝ї
        if (newCapacity - minCapacity < 0)
            newCapacity = minCapacity;
        //тєЇТБђТЪЦТќ░т«╣жЄЈТў»тљдУХЁтЄ║С║єArrayListТЅђт«џС╣ЅуџёТюђтцДт«╣жЄЈ№╝ї
        //УІЦУХЁтЄ║С║є№╝їтѕЎУ░ЃућеhugeCapacity()ТЮЦТ»ћУЙЃminCapacityтњї MAX_ARRAY_SIZE№╝ї
        //тдѓТъюminCapacityтцДС║јТюђтцДт«╣жЄЈ№╝їтѕЎТќ░т«╣жЄЈтѕЎСИ║ArrayListт«џС╣ЅуџёТюђтцДт«╣жЄЈ№╝їтљдтѕЎ№╝їТќ░т«╣жЄЈтцДт░ЈтѕЎСИ║ minCapacityсђѓ 
        if (newCapacity - MAX_ARRAY_SIZE > 0)
            newCapacity = hugeCapacity(minCapacity);
        // minCapacity is usually close to size, so this is a win:
        elementData = Arrays.copyOf(elementData, newCapacity);
    }
    //Т»ћУЙЃminCapacityтњї MAX_ARRAY_SIZE
    private static int hugeCapacity(int minCapacity) {
        if (minCapacity < 0) // overflow
            throw new OutOfMemoryError();
        return (minCapacity > MAX_ARRAY_SIZE) ?
            Integer.MAX_VALUE :
            MAX_ARRAY_SIZE;
    }

    /**
     *У┐ћтЏъТГцтѕЌУАеСИГуџётЁЃу┤аТЋ░сђѓ 
     */
    public int size() {
        return size;
    }

    /**
     * тдѓТъюТГцтѕЌУАеСИЇтїЁтљФтЁЃу┤а№╝їтѕЎУ┐ћтЏъ true сђѓ
     */
    public boolean isEmpty() {
        //Т│еТёЈ=тњї==уџётї║тѕФ
        return size == 0;
    }

    /**
     * тдѓТъюТГцтѕЌУАетїЁтљФТїЄт«џуџётЁЃу┤а№╝їтѕЎУ┐ћтЏъtrue сђѓ
