# тЁѕС╗ј MySQL уџётЪ║ТюгтГўтѓеу╗ЊТъёУ»┤Ухи

MySQLуџётЪ║ТюгтГўтѓеу╗ЊТъёТў»жАх(У«░тйЋжЃйтГўтюежАхжЄїУЙ╣)№╝џ

  • тљёСИфТЋ░ТЇ«жАхтЈ»С╗Цу╗ёТѕљСИђСИфтЈїтљЉжЊЙУАе
  • Т»ЈСИфТЋ░ТЇ«жАхСИГуџёУ«░тйЋтЈѕтЈ»С╗Цу╗ёТѕљСИђСИфтЇЋтљЉжЊЙУАе
    • Т»ЈСИфТЋ░ТЇ«жАхжЃйС╝џСИ║тГўтѓетюет«ЃжЄїУЙ╣тё┐уџёУ«░тйЋућЪТѕљСИђСИфжАхуЏ«тйЋ№╝їтюежђџУ┐ЄСИ╗жћ«ТЪЦТЅЙТЪљТЮАУ«░тйЋуџёТЌХтђЎтЈ»С╗ЦтюежАхуЏ«тйЋСИГСй┐ућеС║їтѕєТ│Ћт┐ФжђЪт«џСйЇтѕ░т»╣т║ћуџёТДй№╝їуёХтљјтєЇжЂЇтјєУ»ЦТДйт»╣т║ћтѕєу╗ёСИГуџёУ«░тйЋтЇ│тЈ»т┐ФжђЪТЅЙтѕ░ТїЄт«џуџёУ«░тйЋ
    • С╗ЦтЁХС╗ќтѕЌ(жЮъСИ╗жћ«)СйюСИ║Тљюу┤бТЮАС╗Х№╝џтЈфУЃйС╗јТюђт░ЈУ«░тйЋт╝ђтДІСЙЮТгАжЂЇтјєтЇЋжЊЙУАеСИГуџёТ»ЈТЮАУ«░тйЋсђѓ

ТЅђС╗ЦУ»┤№╝їтдѓТъюТѕЉС╗гтєЎselect * from user where indexname = 'xxx'У┐ЎТаиТ▓АТюЅУ┐ЏУАїС╗╗СйЋС╝ўтїќуџёsqlУ»ГтЈЦ№╝їж╗ўУ«цС╝џУ┐ЎТаитЂџ№╝џ

  1. т«џСйЇтѕ░У«░тйЋТЅђтюеуџёжАх:жюђУдЂжЂЇтјєтЈїтљЉжЊЙУАе№╝їТЅЙтѕ░ТЅђтюеуџёжАх
  2. С╗јТЅђтюеуџёжАхтєЁСИГТЪЦТЅЙуЏИт║ћуџёУ«░тйЋ:ућ▒С║јСИЇТў»Та╣ТЇ«СИ╗жћ«ТЪЦУ»б№╝їтЈфУЃйжЂЇтјєТЅђтюежАхуџётЇЋжЊЙУАеС║є

тЙѕТўјТўЙ№╝їтюеТЋ░ТЇ«жЄЈтЙѕтцДуџёТЃЁтєхСИІУ┐ЎТаиТЪЦТЅЙС╝џтЙѕТЁб№╝ЂУ┐ЎТаиуџёТЌХжЌ┤тцЇТЮѓт║дСИ║O№╝ѕn№╝Ѕсђѓ

# Сй┐ућеу┤бт╝ЋС╣Ітљј

у┤бт╝ЋтЂџС║єС║ЏС╗ђС╣ѕтЈ»С╗ЦУ«ЕТѕЉС╗гТЪЦУ»бтіат┐ФжђЪт║дтЉб№╝ЪтЁХт«ът░▒Тў»т░єТЌат║ЈуџёТЋ░ТЇ«тЈўТѕљТюЅт║Ј(уЏИт»╣)№╝џ

УдЂТЅЙтѕ░idСИ║8уџёУ«░тйЋу«ђУдЂТГЦжфц№╝џ

тЙѕТўјТўЙуџёТў»№╝џТ▓АТюЅућеу┤бт╝ЋТѕЉС╗гТў»жюђУдЂжЂЇтјєтЈїтљЉжЊЙУАеТЮЦт«џСйЇт»╣т║ћуџёжАх№╝їуј░тюежђџУ┐Є РђюуЏ«тйЋРђЮ т░▒тЈ»С╗ЦтЙѕт┐Фтю░т«џСйЇтѕ░т»╣т║ћуџёжАхСИіС║є№╝Ђ№╝ѕС║їтѕєТЪЦТЅЙ№╝їТЌХжЌ┤тцЇТЮѓт║дУ┐ЉС╝╝СИ║O(logn)№╝Ѕ

тЁХт«ът║Ћт▒ѓу╗ЊТъёт░▒Тў»B+ТаЉ№╝їB+ТаЉСйюСИ║ТаЉуџёСИђуДЇт«ъуј░№╝їУЃйтцЪУ«ЕТѕЉС╗гтЙѕт┐Фтю░ТЪЦТЅЙтЄ║т»╣т║ћуџёУ«░тйЋсђѓ

# ТюђтидтЅЇу╝ђтјЪтѕЎ

MySQLСИГуџёу┤бт╝ЋтЈ»С╗ЦС╗ЦСИђт«џжА║т║Јт╝ЋућетцџтѕЌ№╝їУ┐ЎуДЇу┤бт╝ЋтЈФСйюУЂћтљѕу┤бт╝ЋсђѓтдѓUserУАеуџёnameтњїcityтіаУЂћтљѕу┤бт╝Ћт░▒Тў»(name,city)oУђїТюђтидтЅЇу╝ђтјЪтѕЎТїЄуџёТў»№╝їтдѓТъюТЪЦУ»буџёТЌХтђЎТЪЦУ»бТЮАС╗Ху▓ЙуА«тї╣жЁЇу┤бт╝ЋуџётидУЙ╣У┐ъу╗ГСИђтѕЌТѕќтЄатѕЌ№╝їтѕЎТГцтѕЌт░▒тЈ»С╗ЦУбФућетѕ░сђѓтдѓСИІ№╝џ

select * from user where name=xx and city=xx ; №╝Ј№╝ЈтЈ»С╗ЦтЉйСИГу┤бт╝Ћ
select * from user where name=xx ; // тЈ»С╗ЦтЉйСИГу┤бт╝Ћ
select * from user where city=xx; // ТЌаТ│ЋтЉйСИГу┤бт╝Ћ

У┐ЎжЄїжюђУдЂТ│еТёЈуџёТў»№╝їТЪЦУ»буџёТЌХтђЎтдѓТъюСИцСИфТЮАС╗ХжЃйућеСИіС║є№╝їСйєТў»жА║т║ЈСИЇтљї№╝їтдѓ city= xx and name №╝Юxx№╝їжѓБС╣ѕуј░тюеуџёТЪЦУ»бт╝ЋТЊјС╝џУЄфтіеС╝ўтїќСИ║тї╣жЁЇУЂћтљѕу┤бт╝ЋуџёжА║т║Ј№╝їУ┐ЎТаиТў»УЃйтцЪтЉйСИГу┤бт╝Ћуџё.

ућ▒С║јТюђтидтЅЇу╝ђтјЪтѕЎ№╝їтюетѕЏт╗║УЂћтљѕу┤бт╝ЋТЌХ№╝їу┤бт╝ЋтГЌТ«хуџёжА║т║ЈжюђУдЂУђЃУЎЉтГЌТ«хтђ╝тј╗жЄЇС╣ІтљјуџёСИфТЋ░№╝їУЙЃтцџуџёТћЙтЅЇжЮбсђѓORDERBYтГљтЈЦС╣ЪжЂхтЙфТГцУДётѕЎсђѓ

# Т│еТёЈжЂ┐тЁЇтєЌСйЎу┤бт╝Ћ

тєЌСйЎу┤бт╝ЋТїЄуџёТў»у┤бт╝ЋуџётіЪУЃйуЏИтљї№╝їУЃйтцЪтЉйСИГ т░▒Уѓ»т«џУЃйтЉйСИГ №╝їжѓБС╣ѕ т░▒Тў»тєЌСйЎу┤бт╝Ћтдѓ№╝ѕname,city №╝Ѕтњї№╝ѕname №╝ЅУ┐ЎСИцСИфу┤бт╝Ћт░▒Тў»тєЌСйЎу┤бт╝Ћ№╝їУЃйтцЪтЉйСИГтљјУђЁуџёТЪЦУ»бУѓ»т«џТў»УЃйтцЪтЉйСИГтЅЇУђЁуџё тюетцДтцџТЋ░ТЃЁтєхСИІ№╝їжЃйт║ћУ»Цт░йжЄЈТЅЕт▒Ћти▓ТюЅуџёу┤бт╝ЋУђїСИЇТў»тѕЏт╗║Тќ░у┤бт╝Ћсђѓ

# MysqlтдѓСйЋСИ║УАетГЌТ«хТи╗тіау┤бт╝Ћ№╝Ъ№╝Ъ№╝Ъ

1.Ти╗тіаPRIMARY KEY№╝ѕСИ╗жћ«у┤бт╝Ћ№╝Ѕ

ALTER TABLE `table_name` ADD PRIMARY KEY ( `column` )

2.Ти╗тіаUNIQUE(тћ»СИђу┤бт╝Ћ)

ALTER TABLE `table_name` ADD UNIQUE ( `column` )

3.Ти╗тіаINDEX(ТЎ«жђџу┤бт╝Ћ)

ALTER TABLE `table_name` ADD INDEX index_name ( `column` )

4.Ти╗тіаFULLTEXT(тЁеТќЄу┤бт╝Ћ)

ALTER TABLE `table_name` ADD FULLTEXT ( `column`)

5.Ти╗тіатцџтѕЌу┤бт╝Ћ

ALTER TABLE `table_name` ADD INDEX index_name ( `column1`, `column2`, `column3` )
