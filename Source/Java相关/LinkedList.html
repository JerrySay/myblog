        Node<E> pred, succ;

        if (index == size) {
            succ = null;
            pred = last;

        //тљдтѕЎ№╝їУ░Ѓућеnode()Тќ╣Т│ЋтЙЌтѕ░тљју╗ДУіѓуѓ╣№╝їтєЇтЙЌтѕ░тЅЇжЕ▒Уіѓуѓ╣
        else {
            succ = node(index);
            pred = succ.prev;
        }

        // 4№╝џжЂЇтјєТЋ░ТЇ«т░єТЋ░ТЇ«ТЈњтЁЦ
        for (Object o : a) {
            @SuppressWarnings("unchecked") E e = (E) o;
            //тѕЏт╗║Тќ░Уіѓуѓ╣
            Node<E> newNode = new Node<>(pred, e, null);
            //тдѓТъюТЈњтЁЦСйЇуй«тюежЊЙУАетц┤жЃе
            if (pred == null)

        modCount++;
        return true;
    }

            //С╗јтц┤жЂЇтјє

    }

            x.prev = null;

    }
