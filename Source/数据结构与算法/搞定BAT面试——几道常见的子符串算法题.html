тЅЉТїЄoffer№╝џУ»ит«ъуј░СИђСИфтЄйТЋ░№╝їт░єСИђСИфтГЌугдСИ▓СИГуџёТ»ЈСИфуЕ║Та╝ТЏ┐ТЇбТѕљРђю%20РђЮсђѓСЙІтдѓ№╝їтйЊтГЌугдСИ▓СИ║We Are Happy.тѕЎу╗ЈУ┐ЄТЏ┐ТЇбС╣ІтљјуџётГЌугдСИ▓СИ║We%20Are%20Happyсђѓ

У┐ЎжЄїТѕЉТЈљСЙЏС║єСИцуДЇТќ╣Т│Ћ№╝џРЉатИИУДёТќ╣Т│Ћ№╝ЏРЉАтѕЕуће API УДБтє│сђѓ

//https://www.weiweiblog.cn/replacespace/
public class Solution {

  /**
   * уггСИђуДЇТќ╣Т│Ћ№╝џтИИУДёТќ╣Т│ЋсђѓтѕЕућеString.charAt(i)С╗ЦтЈіString.valueOf(char).equals(" "
   * )жЂЇтјєтГЌугдСИ▓т╣ХтѕцТќГтЁЃу┤аТў»тљдСИ║уЕ║Та╝сђѓТў»тѕЎТЏ┐ТЇбСИ║"%20",тљдтѕЎСИЇТЏ┐ТЇб
   */
  public static String replaceSpace(StringBuffer str) {

    int length = str.length();
    // System.out.println("length=" + length);
    StringBuffer result = new StringBuffer();
    for (int i = 0; i < length; i++) {
      char b = str.charAt(i);
      if (String.valueOf(b).equals(" ")) {
        result.append("%20");
      } else {
        result.append(b);
      }
    }
    return result.toString();

  }

  /**
   * уггС║їуДЇТќ╣Т│Ћ№╝џтѕЕућеAPIТЏ┐ТЇбТјЅТЅђућеуЕ║Та╝№╝їСИђУАїС╗БуаЂУДБтє│жЌ«жбў
   */
  public static String replaceSpace2(StringBuffer str) {

    return str.toString().replaceAll("\\s", "%20");
  }
}

# 3. ТюђжЋ┐тЁгтЁ▒тЅЇу╝ђ

Leetcode: у╝ќтєЎСИђСИфтЄйТЋ░ТЮЦТЪЦТЅЙтГЌугдСИ▓ТЋ░у╗ёСИГуџёТюђжЋ┐тЁгтЁ▒тЅЇу╝ђсђѓтдѓТъюСИЇтГўтюетЁгтЁ▒тЅЇу╝ђ№╝їУ┐ћтЏъуЕ║тГЌугдСИ▓ ""сђѓ

//https://leetcode-cn.com/problems/longest-common-prefix/description/
public class Main {
  public static String replaceSpace(String[] strs) {

    // ТЋ░у╗ёжЋ┐т║д
    int len = strs.length;
    // ућеС║јС┐ЮтГўу╗ЊТъю
    StringBuffer res = new StringBuffer();
    // Т│еТёЈ№╝џ=Тў»УхІтђ╝№╝ї==Тў»тѕцТќГ
    if (strs == null || strs.length == 0) {
      return "";
    }
    // у╗ЎтГЌугдСИ▓ТЋ░у╗ёуџётЁЃу┤аТїЅуЁДтЇЄт║ЈТјњт║Ј(тїЁтљФТЋ░тГЌуџёУ»Ю№╝їТЋ░тГЌС╝џТјњтюетЅЇжЮб)
    Arrays.sort(strs);
    int m = strs[0].length();
    int n = strs[len - 1].length();
    int num = Math.min(m, n);
    for (int i = 0; i < num; i++) {
      if (strs[0].charAt(i) == strs[len - 1].charAt(i)) {
        res.append(strs[0].charAt(i));
      } else
        break;

    }
    return res.toString();

  } 
     //ТхІУ»Ћ
  public static void main(String[] args) {
    String[] strs = { "customer", "car", "cat" };
    System.out.println(Main.replaceSpace(strs));//c
  }
}

class Solution {
  public  int longestPalindrome(String s) {
    if (s.length() == 0)
      return 0;
    // ућеС║јтГўТћЙтГЌугд
    HashSet<Character> hashset = new HashSet<Character>();
    char[] chars = s.toCharArray();
    int count = 0;
    for (int i = 0; i < chars.length; i++) {
      if (!hashset.contains(chars[i])) {// тдѓТъюhashsetТ▓АТюЅУ»ЦтГЌугдт░▒С┐ЮтГўУ┐Џтј╗
        hashset.add(chars[i]);
      } else {// тдѓТъюТюЅ,т░▒У«Еcount++№╝ѕУ»┤ТўјТЅЙтѕ░С║єСИђСИфТѕљт»╣уџётГЌугд№╝Ѕ№╝їуёХтљјТііУ»ЦтГЌугдуД╗жЎц
        hashset.remove(chars[i]);
        count++;
      }
    }
    return hashset.isEmpty() ? count * 2 : count * 2 + 1;
  }
}

class Solution {
  public  boolean isPalindrome(String s) {
    if (s.length() == 0)
      return true;
    int l = 0, r = s.length() - 1;
    while (l < r) {
      // С╗јтц┤тњїт░Йт╝ђтДІтљЉСИГжЌ┤жЂЇтјє
      if (!Character.isLetterOrDigit(s.charAt(l))) {// тГЌугдСИЇТў»тГЌТ»ЇтњїТЋ░тГЌуџёТЃЁтєх
        l++;
      } else if (!Character.isLetterOrDigit(s.charAt(r))) {// тГЌугдСИЇТў»тГЌТ»ЇтњїТЋ░тГЌуџёТЃЁтєх
        r--;
      } else {
        // тѕцТќГС║їУђЁТў»тљдуЏИуГЅ
        if (Character.toLowerCase(s.charAt(l)) != Character.toLowerCase(s.charAt(r)))
          return false;
        l++;
        r--;
      }
    }
    return true;
  }
}

class Solution {
  private int index, len;

  public String longestPalindrome(String s) {
    if (s.length() < 2)
      return s;
    for (int i = 0; i < s.length() - 1; i++) {
      PalindromeHelper(s, i, i);
      PalindromeHelper(s, i, i + 1);
    }
    return s.substring(index, index + len);
  }

  public void PalindromeHelper(String s, int l, int r) {
    while (l >= 0 && r < s.length() && s.charAt(l) == s.charAt(r)) {
      l--;
      r++;
    }
    if (len < r - l - 1) {
      index = l + 1;
      len = r - l - 1;
    }
  }
}

class Solution {
    public int longestPalindromeSubseq(String s) {
        int len = s.length();
        int [][] dp = new int[len][len];
        for(int i = len - 1; i>=0; i--){
            dp[i][i] = 1;
            for(int j = i+1; j < len; j++){
                if(s.charAt(i) == s.charAt(j))
                    dp[i][j] = dp[i+1][j-1] + 2;
                else
                    dp[i][j] = Math.max(dp[i+1][j], dp[i][j-1]);
            }
        }
        return dp[0][len-1];
    }
}
