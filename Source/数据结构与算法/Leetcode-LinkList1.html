/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
 //https://leetcode-cn.com/problems/add-two-numbers/description/
class Solution {
public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
    ListNode dummyHead = new ListNode(0);
    ListNode p = l1, q = l2, curr = dummyHead;
    //carry УАеуц║У┐ЏСйЇТЋ░
    int carry = 0;
    while (p != null || q != null) {
        int x = (p != null) ? p.val : 0;
        int y = (q != null) ? q.val : 0;
        int sum = carry + x + y;
        //У┐ЏСйЇТЋ░
        carry = sum / 10;
        //Тќ░Уіѓуѓ╣уџёТЋ░тђ╝СИ║sum % 10
        curr.next = new ListNode(sum % 10);
        curr = curr.next;
        if (p != null) p = p.next;
        if (q != null) q = q.next;
    }
    if (carry > 0) {
        curr.next = new ListNode(carry);
    }
    return dummyHead.next;
}
}

  public ListNode ReverseList(ListNode head) {

    ListNode next = null;
    ListNode pre = null;

    while (head != null) {
      // С┐ЮтГўУдЂтЈЇУйгтѕ░тц┤уџёжѓБСИфУіѓуѓ╣
      next = head.next;

      pre = head;
      // СИђуЏ┤тљЉжЊЙУАет░ЙУх░
      head = next;
    }
    return pre;
  }

}

# 3. жЊЙУАеСИГтђњТЋ░уггkСИфУіѓуѓ╣

# жбўуЏ«ТЈЈУ┐░

тЅЉТїЄoffer: УЙЊтЁЦСИђСИфжЊЙУАе№╝їУЙЊтЄ║У»ЦжЊЙУАеСИГтђњТЋ░уггkСИфу╗Њуѓ╣сђѓ

# жЌ«жбўтѕєТъљ

жЊЙУАеСИГтђњТЋ░уггkСИфУіѓуѓ╣С╣Ът░▒Тў»ТГБТЋ░угг(L-K+1)СИфУіѓуѓ╣№╝їуЪЦжЂЊС║єтЈфСИђуѓ╣№╝їУ┐ЎСИђжбўтЪ║Тюгт░▒Т▓АжЌ«жбў№╝Ђ

ждќтЁѕСИцСИфУіѓуѓ╣/ТїЄжњѕ№╝їСИђСИфУіѓуѓ╣ node1 тЁѕт╝ђтДІУиЉ№╝їТїЄжњѕ node1 УиЉтѕ░ k-1 СИфУіѓуѓ╣тљј№╝їтЈдСИђСИфУіѓуѓ╣ node2 т╝ђтДІУиЉ№╝їтйЊ node1 УиЉтѕ░ТюђтљјТЌХ№╝їnode2 ТЅђТїЄуџёУіѓуѓ╣т░▒Тў»тђњТЋ░уггkСИфУіѓуѓ╣С╣Ът░▒Тў»ТГБТЋ░угг(L-K+1)СИфУіѓуѓ╣сђѓ

// ТЌХжЌ┤тцЇТЮѓт║дO(n),СИђТгАжЂЇтјєтЇ│тЈ»
// https://www.nowcoder.com/practice/529d3ae5a407492994ad2a246518148a?tpId=13&tqId=11167&tPage=1&rp=1&ru=/ta/coding-interviews&qru=/ta/coding-interviews/question-ranking
public class Solution {
  public ListNode FindKthToTail(ListNode head, int k) {
    // тдѓТъюжЊЙУАеСИ║уЕ║ТѕќУђЁkт░ЈС║јуГЅС║ј0
    if (head == null || k <= 0) {
      return null;
    }
    // тБ░ТўјСИцСИфТїЄтљЉтц┤у╗Њуѓ╣уџёУіѓуѓ╣
    ListNode node1 = head, node2 = head;
    // У«░тйЋУіѓуѓ╣уџёСИфТЋ░
    int count = 0;
    // У«░тйЋkтђ╝№╝їтљјжЮбУдЂСй┐уће
    int index = k;
    // pТїЄжњѕтЁѕУиЉ№╝їт╣ХСИћУ«░тйЋУіѓуѓ╣ТЋ░№╝їтйЊnode1Уіѓуѓ╣УиЉС║єk-1СИфУіѓуѓ╣тљј№╝їnode2Уіѓуѓ╣т╝ђтДІУиЉ№╝ї
    // тйЊnode1Уіѓуѓ╣УиЉтѕ░ТюђтљјТЌХ№╝їnode2Уіѓуѓ╣ТЅђТїЄуџёУіѓуѓ╣т░▒Тў»тђњТЋ░уггkСИфУіѓуѓ╣
    while (node1 != null) {
      node1 = node1.next;
      count++;
      if (k < 1 && node1 != null) {
        node2 = node2.next;
      }
      k--;
    }
    // тдѓТъюУіѓуѓ╣СИфТЋ░т░ЈС║јТЅђТ▒ѓуџётђњТЋ░уггkСИфУіѓуѓ╣№╝їтѕЎУ┐ћтЏъуЕ║
    if (count < index)
      return null;
    return node2;

  }
}

// https://leetcode-cn.com/problems/remove-nth-node-from-end-of-list/description/
public class Solution {
  public ListNode removeNthFromEnd(ListNode head, int n) {
    // тЊЉу╗Њуѓ╣№╝їтЊЉу╗Њуѓ╣ућеТЮЦу«ђтїќТЪљС║ЏТъЂуФ»ТЃЁтєх№╝їСЙІтдѓтѕЌУАеСИГтЈфтљФТюЅСИђСИфу╗Њуѓ╣№╝їТѕќжюђУдЂтѕажЎцтѕЌУАеуџётц┤жЃе
    ListNode dummy = new ListNode(0);
    // тЊЉу╗Њуѓ╣ТїЄтљЉтц┤у╗Њуѓ╣
    dummy.next = head;
    // С┐ЮтГўжЊЙУАежЋ┐т║д
    int length = 0;
    ListNode len = head;
    while (len != null) {
      length++;
      len = len.next;
    }
    length = length - n;
    ListNode target = dummy;
    // ТЅЙтѕ░ L-n СйЇуй«уџёУіѓуѓ╣
    while (length > 0) {
      target = target.next;
      length--;
    }
    // Тііугг (L - n)СИфу╗Њуѓ╣уџё next ТїЄжњѕжЄЇТќ░жЊЙТјЦУЄ│угг (L - n + 2)СИфу╗Њуѓ╣
    target.next = target.next.next;
    return dummy.next;
  }
}

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
public class Solution {
  public ListNode removeNthFromEnd(ListNode head, int n) {

    ListNode dummy = new ListNode(0);
    dummy.next = head;
    // тБ░ТўјСИцСИфТїЄтљЉтц┤у╗Њуѓ╣уџёУіѓуѓ╣
    ListNode node1 = dummy, node2 = dummy;

    // node1 Уіѓуѓ╣тЁѕУиЉ№╝їnode1Уіѓуѓ╣ УиЉтѕ░угг n СИфУіѓуѓ╣уџёТЌХтђЎ,node2 Уіѓуѓ╣т╝ђтДІУиЉ
    // тйЊnode1 Уіѓуѓ╣УиЉтѕ░ТюђтљјСИђСИфУіѓуѓ╣ТЌХ№╝їnode2 Уіѓуѓ╣ТЅђтюеуџёСйЇуй«т░▒Тў»угг №╝ѕL-n №╝Ѕ СИфУіѓуѓ╣№╝їС╣Ът░▒Тў»тђњТЋ░угг n+1№╝ѕLС╗БУАеТђ╗жЊЙУАежЋ┐т║д№╝Ѕ
    while (node1 != null) {
      node1 = node1.next;
      if (n < 1 && node1 != null) {
        node2 = node2.next;
      }
      n--;
    }

    node2.next = node2.next.next;

    return dummy.next;

  }
}
