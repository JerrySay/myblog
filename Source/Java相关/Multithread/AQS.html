public class SemaphoreExample1 {

  private static final int threadCount = 550;

  public static void main(String[] args) throws InterruptedException {
    // тѕЏт╗║СИђСИфтЁиТюЅтЏ║т«џу║┐уеІТЋ░жЄЈуџёу║┐уеІТ▒ат»╣У▒А№╝ѕтдѓТъюУ┐ЎжЄїу║┐уеІТ▒ауџёу║┐уеІТЋ░жЄЈу╗Ўтцфт░ЉуџёУ»ЮСйаС╝џтЈЉуј░ТЅДУАїуџётЙѕТЁб№╝Ѕ

    final Semaphore semaphore = new Semaphore(20);

    for (int i = 0; i < threadCount; i++) {
      final int threadnum = i;
      threadPool.execute(() -> {// Lambda УАеУЙЙт╝ЈуџёУ┐љуће
        try {
          semaphore.acquire();// УјитЈќСИђСИфУ«ИтЈ»№╝їТЅђС╗ЦтЈ»У┐љУАїу║┐уеІТЋ░жЄЈСИ║20/1=20
          test(threadnum);
          semaphore.release();// жЄіТћЙСИђСИфУ«ИтЈ»
        } catch (InterruptedException e) {
          // TODO Auto-generated catch block
          e.printStackTrace();
        }

      });
    }
    threadPool.shutdown();
    System.out.println("finish");
  }

  }
}

  // У»иТ▒ѓуџёТЋ░жЄЈ
  private static final int threadCount = 550;

  public static void main(String[] args) throws InterruptedException {
    // тѕЏт╗║СИђСИфтЁиТюЅтЏ║т«џу║┐уеІТЋ░жЄЈуџёу║┐уеІТ▒ат»╣У▒А№╝ѕтдѓТъюУ┐ЎжЄїу║┐уеІТ▒ауџёу║┐уеІТЋ░жЄЈу╗Ўтцфт░ЉуџёУ»ЮСйаС╝џтЈЉуј░ТЅДУАїуџётЙѕТЁб№╝Ѕ

      final int threadnum = i;
      threadPool.execute(() -> {// Lambda УАеУЙЙт╝ЈуџёУ┐љуће

        }

      });
    }

  }
